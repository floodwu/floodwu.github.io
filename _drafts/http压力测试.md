# 在Mac上安装webbench
```
# 安装依赖
brew install ctags

cd /Users/wujian/extensions
wget http://blog.zyan.cc/soft/linux/webbench/webbench-1.5.tar.gz
tar -zxvf webbench-1.5.tar.gz
cd webbench-1.5
sudo mkdir -pv /usr/local/man/man1
sudo make && sudo make install
```
